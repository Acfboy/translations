题目要求 $N$ 个数的和，我们从 $1$ 枚举到 $N$，把所有数加起来即可。

```cpp
#include<bits/stdc++.h>
using namespace std;
int n,a[105],s;
signed main(){
    cin>>n;
    for(int i=1;i<=n;i++)cin>>a[i],s+=a[i];
    cout<<s<<'\n';
    return 0;
}
```
